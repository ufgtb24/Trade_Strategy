```python
"""渲染 samples/<id>/chart.png — AI 友好的静态 K 线图。

复用 BreakoutStrategy/UI/charts/components/candlestick.py 的
CandlestickComponent.draw 在 matplotlib Axes 上绘制 K 线主体；
本模块负责构造 Figure、添加 volume subplot、标注 bo / pk 位置、
导出 PNG（不交互、固定尺寸、统一风格）。

OO API 设计：使用 matplotlib.figure.Figure + FigureCanvasAgg 直接渲染，
不调 matplotlib.use() / pyplot —— 避免污染调用方进程的全局 backend
（dev UI 的 TkAgg 同进程调用此模块时尤其关键）。
"""

from pathlib import Path

import pandas as pd
from matplotlib.figure import Figure
from matplotlib.backends.backend_agg import FigureCanvasAgg

from BreakoutStrategy.feature_library import paths
from BreakoutStrategy.UI.charts.components.candlestick import CandlestickComponent

CHART_DPI = 100
CHART_FIGSIZE = (12, 8)
PRICE_PANEL_RATIO = 3
VOLUME_PANEL_RATIO = 1


def render_sample_chart(
    sample_id: str,
    df_window: pd.DataFrame,
    bo_index: int,
    pk_index: int,
) -> Path:
    """渲染单个 breakout 样本的 K 线图为 PNG。

    Args:
        sample_id: 样本 ID
        df_window: OHLCV DataFrame，DatetimeIndex
        bo_index: 突破日 index（df_window 局部）
        pk_index: 盘整起点 index（df_window 局部）

    Returns:
        生成的 PNG 文件绝对路径
    """
    paths.ensure_sample_dir(sample_id)
    out_path = paths.chart_png_path(sample_id)

    # OO API：不进 pyplot 全局 figure 表，不切换全局 backend
    fig = Figure(figsize=CHART_FIGSIZE, dpi=CHART_DPI)
    canvas = FigureCanvasAgg(fig)

    ax_price = fig.add_subplot(2, 1, 1)
    ax_vol = fig.add_subplot(2, 1, 2, sharex=ax_price)
    fig.subplots_adjust(hspace=0.05)
    # 通过 gridspec_kw 调整高度比例
    gs = fig.add_gridspec(
        2, 1, height_ratios=[PRICE_PANEL_RATIO, VOLUME_PANEL_RATIO]
    )
    ax_price.set_position(gs[0].get_position(fig))
    ax_vol.set_position(gs[1].get_position(fig))

    # 主图：K 线（CandlestickComponent.draw 接受任何 Axes）
    CandlestickComponent.draw(ax_price, df_window)

    # 标注 pk 与 bo 垂直线
    ax_price.axvline(
        x=pk_index, color="#FF9800", linestyle="--",
        linewidth=1.5, label="pk (consolidation start)",
    )
    ax_price.axvline(
        x=bo_index, color="#2196F3", linestyle="--",
        linewidth=1.5, label="bo (breakout)",
    )
    ax_price.set_title(f"{sample_id}")
    ax_price.set_ylabel("Price")
    ax_price.legend(loc="upper left")

    # 副图：volume bar
    ax_vol.bar(
        range(len(df_window)), df_window["volume"], color="#888888", width=0.8,
    )
    ax_vol.axvline(x=pk_index, color="#FF9800", linestyle="--", linewidth=1.0)
    ax_vol.axvline(x=bo_index, color="#2196F3", linestyle="--", linewidth=1.0)
    ax_vol.set_ylabel("Volume")
    ax_vol.set_xlabel("Bar Index")
    ax_vol.grid(True, alpha=0.3)

    # 直接通过 canvas 写文件，不走 pyplot.savefig（避免污染全局 figure 表）
    canvas.print_png(out_path)
    return out_path
```

def _build_figure_for_inspection(
    df_window: pd.DataFrame, bo_index: int, pk_index: int,
) -> Figure:
    """构造脱敏 K 线图 Figure（内部工具，单元测试用）。

    与 render_sample_chart 共享渲染逻辑，但不落盘——便于测试断言图层属性。
    """
    fig = Figure(figsize=CHART_FIGSIZE, dpi=CHART_DPI)
    FigureCanvasAgg(fig)

    gs = fig.add_gridspec(
        2, 1, height_ratios=[PRICE_PANEL_RATIO, VOLUME_PANEL_RATIO],
        hspace=0.05,
    )
    ax_price = fig.add_subplot(gs[0])
    ax_vol = fig.add_subplot(gs[1], sharex=ax_price)

    # K 线主图
    CandlestickComponent.draw(ax_price, df_window)

    # pk / bo 垂直线
    ax_price.axvline(
        x=pk_index, color="#FF9800", linestyle="--", linewidth=1.5,
        label="pk (consolidation start)",
    )
    ax_price.axvline(
        x=bo_index, color="#2196F3", linestyle="--", linewidth=1.5,
        label="bo (breakout)",
    )

    # 脱敏：标题改为通用文本（不含 ticker/sample_id/日期）
    ax_price.set_title("Breakout sample (anonymized)")
    ax_price.set_ylabel("Price (% vs pk)")
    ax_price.legend(loc="upper left")

    # Y 轴用 pk_close 作为 pivot，FuncFormatter 显示相对 %
    pivot_close = float(df_window.iloc[pk_index]["close"])

    def _pct_fmt(price: float, _pos: int) -> str:
        if pivot_close <= 0:
            return f"{price:.2f}"
        pct = (price - pivot_close) / pivot_close * 100
        sign = "+" if pct > 0 else ""
        return f"{sign}{pct:.1f}%"

    ax_price.yaxis.set_major_formatter(FuncFormatter(_pct_fmt))

    # 副图：volume bar
    ax_vol.bar(
        range(len(df_window)), df_window["volume"], color="#888888", width=0.8,
    )
    ax_vol.axvline(x=pk_index, color="#FF9800", linestyle="--", linewidth=1.0)
    ax_vol.axvline(x=bo_index, color="#2196F3", linestyle="--", linewidth=1.0)
    ax_vol.set_ylabel("Volume")
    ax_vol.set_xlabel("Bar Index")
    ax_vol.grid(True, alpha=0.3)

    return fig

```python
def build_user_message(meta: dict[str, Any]) -> str:
    """构造发送给 GLM-4V-Flash 的 user 文本（与图像同消息）。

    归一化方案 B：脱敏 ticker / bo_date / 绝对价位与成交量。
    OHLCV 改为相对 consolidation.pivot_close 的百分比。
    若 meta.consolidation.pivot_close 缺失则跳过 OHLCV 段（降级行为）。
    5 个盘整字段已是无量纲（%、比值、bar 数），保留原值。

    Args:
        meta: sample_meta dict（落盘 meta.yaml 仍含原始 ticker/bo_date/价位
              用于人类追溯，本函数仅在内存中脱敏 prompt 副本）。

    Returns:
        脱敏后的纯文本字符串。
    """
    bo = meta["breakout_day"]
    consol = meta["consolidation"]
    pivot_close = consol.get("pivot_close")

    def fmt_pct(v) -> str:
        if not isinstance(v, (int, float)):
            return "N/A"
        return f"{v:+.2f}%"

    def fmt_num(v) -> str:

def build_batch_user_message(
    samples_meta: list[dict[str, Any]],
    *,
    return_id_map: bool = False,
):
    """构造发送给 GLM-4V-Flash 的 batch user 文本（与 N 张图同消息）。

    归一化方案 B：每个 sample 用 [1] / [2] 匿名编号；不输出 ticker / bo_date /

                       从 [1]/[2] 反查回真实 sample_id。

    Returns:
        return_id_map=False: 脱敏后的 user message 字符串
        return_id_map=True: (user message, id_map dict)
    """

    def fmt_num(v) -> str:
        if isinstance(v, bool):
            return str(v)
        return f"{v:.2f}" if isinstance(v, (int, float)) else "N/A"

    def fmt_pct(v) -> str:
        if not isinstance(v, (int, float)):
            return "N/A"
        return f"{v:+.2f}%"

    lines = [
        f"我给你 {len(samples_meta)} 张 K 线图，按顺序匿名编号 [1] ~ [{len(samples_meta)}]：\n"
    ]
    id_map: dict[str, str] = {}
    for i, meta in enumerate(samples_meta, start=1):
        anon = f"[{i}]"
        id_map[anon] = meta["sample_id"]
        bo = meta["breakout_day"]
        consol = meta["consolidation"]
        pivot_close = consol.get("pivot_close")
        if isinstance(pivot_close, (int, float)) and pivot_close > 0:

                f"breakout_day_pct(vs pk_close): "
                f"open={fmt_pct((bo['open'] - pivot_close) / pivot_close * 100)} "
                f"high={fmt_pct((bo['high'] - pivot_close) / pivot_close * 100)} "
                f"low={fmt_pct((bo['low'] - pivot_close) / pivot_close * 100)} "
                f"close={fmt_pct((bo['close'] - pivot_close) / pivot_close * 100)}"
            )
        else:
            ohlc_segment = "breakout_day_pct: N/A (pivot_close missing)"

        lines.append(
            f"\n{anon}\n"
            f"    {ohlc_segment}\n"
            f"    consolidation: length={fmt_num(consol['consolidation_length_bars'])} bars, "
            f"height={fmt_num(consol['consolidation_height_pct'])}%, "
            f"vs_52w_high={fmt_num(consol['consolidation_position_vs_52w_high'])}%, "
            f"volume_ratio={fmt_num(consol['consolidation_volume_ratio'])}, "
            f"tightness_atr={fmt_num(consol['consolidation_tightness_atr'])}"
        )
    lines.append("\n\n请按 SYSTEM_PROMPT 要求归纳这批样本的共性规律。")
    msg = "".join(lines)
    if return_id_map:
        return msg, id_map
    return msg

使用例（canvas_manager 内）：
    picker = KeyboardPicker(
        get_hovered_bar=lambda: self._last_hover_x,
        get_bo_bars=lambda: {bo.index for bo in self._current_breakouts},
        on_render=self._trigger_sample_render,
        on_toast=self._show_toast,
        on_sticky=self._set_status_sticky,
        on_marker_redraw=self._redraw_endpoint_markers,
    )
    canvas_widget.bind("<p>", lambda e: picker.on_press_p())
    canvas_widget.bind("<P>", lambda e: picker.on_press_p())
"""

from enum import Enum
from typing import Callable, Iterable, Optional


class PickerStatus(str, Enum):
    IDLE = "idle"
    PICKING_1 = "picking_1"


class KeyboardPicker:
    """两端点挑选状态机，规则：右端 ∈ BO bar 集合（场景 A）。"""

    def __init__(
        self,
        *,
        get_hovered_bar: Callable[[], Optional[int]],
        get_bo_bars: Callable[[], Iterable[int]],
        on_render: Callable[[int, int], None],
        on_toast: Callable[[str, str], None],
        on_sticky: Callable[[str], None],
        on_marker_redraw: Callable[[Iterable[int]], None],
    ) -> None:
        # 注入依赖（便于单元测试）
        self._get_hovered_bar = get_hovered_bar
        self._get_bo_bars = get_bo_bars
        self._on_render = on_render
        self._on_toast = on_toast
        self._on_sticky = on_sticky
        self._on_marker_redraw = on_marker_redraw

        # 状态
        self.status: PickerStatus = PickerStatus.IDLE
        self.endpoint_bars: list[int] = []

    # ---------- 公开 API ----------

    def on_press_p(self) -> None:
        """处理 P 键按下事件。"""
        bar = self._get_hovered_bar()
        if bar is None:
            self._on_toast("info", "Hover over a bar first")
            return

        # 已有第一根时检查是否同一根
        if self.endpoint_bars and self.endpoint_bars[0] == bar:
            self._on_toast("info", "Same bar, hover another")
            return

        self.endpoint_bars.append(bar)

        if len(self.endpoint_bars) == 1:
            # 第一根：进入 picking_1
            self.status = PickerStatus.PICKING_1
            self._on_sticky(
                "📌 Picking sample (1/2) — hover & press P on second bar, ESC to cancel"
            )
            self._on_marker_redraw(list(self.endpoint_bars))
            return

        # 第二根：自动判定
        left_idx, right_idx = sorted(self.endpoint_bars)
        bo_set = set(self._get_bo_bars())
        if right_idx in bo_set:
            # 合法：触发渲染并回到 idle
            self._on_render(left_idx, right_idx)
            self._reset_silent()
        else:
            # 非法：保留第一根，清第二根
            self._on_toast(
                "warning",
                "Right endpoint must be a BO bar (red triangle marker)",
            )
            self.endpoint_bars = [self.endpoint_bars[0]]
            self._on_marker_redraw(list(self.endpoint_bars))

    def on_press_escape(self) -> bool:
        """处理 Esc 键按下事件。

        Returns:
            True 表示 picker 消费了事件（picking 状态被清空）；
            False 表示 idle 状态下 picker 不感兴趣，调用方应透传给其他 handler。
        """
        if self.status == PickerStatus.IDLE:
            return False
        self._reset_silent()
        self._on_toast("info", "Sample picking cancelled")
        return True

    def reset(self) -> None:
        """外部强制重置（如切换 ticker 时调用）。"""
        self._reset_silent()

    # ---------- 内部 ----------

    def _reset_silent(self) -> None:
        """清状态 + 重绘空 marker + 清 sticky；不发 toast。"""
        self.status = PickerStatus.IDLE
        self.endpoint_bars = []
        self._on_marker_redraw([])
        self._on_sticky("")
```

"""dev UI sample picker 集成 handler。

把 KeyboardPicker 挑出的两个端点 (left_idx, right_idx) 转成
feature_library.preprocess.preprocess_sample 的调用，处理：
- 全局 index → df_window 局部 index 转换（200 bar 切片）
- bo_date 推算
- preprocess 异常捕获（不让异常逃逸阻塞 Tk 主循环）
- on_done 回调（sample_id 或 None）

与 dev/main.py 解耦——便于单元测试 + 未来其他入口（如 web UI）复用。
"""

from datetime import datetime
from typing import Callable, Optional

import pandas as pd

from BreakoutStrategy.feature_library.preprocess import preprocess_sample

WINDOW_BARS_BEFORE_BO = 200


def handle_endpoints_picked(
    *,
    left_idx: int,
    right_idx: int,
    ticker: str,
    df: pd.DataFrame,
    picked_at: datetime,
    backend,
    on_done: Callable[[Optional[str]], None],
) -> None:
    """把两个全局 bar index 转成 preprocess_sample 调用。

    Args:
        left_idx: 较小的全局 bar index（pk_index 候选）
        right_idx: 较大的全局 bar index（bo_index 候选；调用方已保证 ∈ BO 集合）
        ticker: 股票代码
        df: 完整 OHLCV DataFrame（dev UI 的 current_df）
        picked_at: 用户挑选时刻
        backend: 多模态 backend（如 GLM4VBackend），需有 describe_chart 方法
        on_done: 完成回调，接收 sample_id 或 None（失败时）
    """
    window_start = max(0, right_idx - WINDOW_BARS_BEFORE_BO)
    df_window = df.iloc[window_start: right_idx + 1]
    local_bo = right_idx - window_start
    local_pk = left_idx - window_start
    bo_date = df.index[right_idx]

    try:
        sample_id = preprocess_sample(
            ticker=ticker,
            bo_date=bo_date,
            df_window=df_window,
            bo_index=local_bo,
            pk_index=local_pk,
            picked_at=picked_at,
            backend=backend,
        )
    except Exception as exc:
        # 捕获以避免 Tk 主循环被异常挂掉；调用方通过 on_done(None) 感知失败
        print(f"[sample_picker_handler] preprocess_sample 失败：{exc!r}")
        on_done(None)
        return

    on_done(sample_id)
```

from pathlib import Path
import yaml

from BreakoutStrategy.feature_library.glm4v_backend import GLM4VBackend

_REPO_ROOT = Path(__file__).resolve().parents[2]
_API_KEYS_PATH = _REPO_ROOT / "configs" / "api_keys.yaml"
_GLM4V_SINGLETON: Optional[GLM4VBackend] = None


def get_glm4v_backend_lazy() -> Optional[GLM4VBackend]:
    """lazy 实例化 GLM4VBackend 单例。

    Returns:
        GLM4VBackend 实例；若 api_keys.yaml 缺失或 zhipuai key 为空则 None
        （调用方应据此禁用 picker 或显示错误提示）。
    """
    global _GLM4V_SINGLETON
    if _GLM4V_SINGLETON is not None:
        return _GLM4V_SINGLETON
    if not _API_KEYS_PATH.exists():
        return None
    try:
        keys = yaml.safe_load(_API_KEYS_PATH.read_text(encoding="utf-8")) or {}
    except Exception as exc:
        print(f"[sample_picker_handler] api_keys.yaml 解析失败：{exc!r}")
        return None
    api_key = keys.get("zhipuai", "").strip()
    if not api_key:
        return None
    _GLM4V_SINGLETON = GLM4VBackend(api_key=api_key)
    return _GLM4V_SINGLETON
```
