## Create a new Git worktree for a branch named "feature" located at "/home/yu/PycharmProjects/worktrees/feature".
# 1. 创建并切换到新的 worktree
git worktree add -b feature /home/yu/PycharmProjects/worktrees/feature
git worktree add -b UI ./worktrees/UI
git commit -m "Add new feature"

# 2. 回到 main worktree
git checkout main

# 3. 合并 feature 分支
git merge feature

# 4. 推送合并后的 main
git push

# 5. 清理不需要的 worktree
git worktree remove ./worktrees/UI

# 6. 删除已合并的分支(可选)
git branch -d feature

------------
## reset the "master" branch to match the remote "origin/master" branch, discarding any local changes.
# 1. 获取远程最新状态
git fetch origin

# 2. 将本地 master 分支强制重置为远程 origin/master 的状态
git reset --hard origin/master
